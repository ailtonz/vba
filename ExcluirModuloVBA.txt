Sub ExcluirModuloVBA()
 
    Dim vbCom As Object
    Dim NomeModulo As String
    NomeModulo = "Módulo1"
    Set vbCom = Application.VBE.ActiveVBProject.VBComponents
 
    vbCom.Remove VBComponent:= vbCom.Item(NomeModulo)
    MsgBox NomeModulo & " excluído com sucesso"
 
End Sub
